## Installation

```bash
npx devcn-ui add toggle
```

## Usage

```tsx
import { Toggle } from '@repo/shadcn-ui/components/toggle';

export default function Example() {
  return (
    <Toggle>
      Your content here
    </Toggle>
  );
}
```

## API Reference

### Toggle

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| `className` | `string` | - | Additional CSS classes |
| `children` | `ReactNode` | - | The content of the component |
