## Installation

```bash
npx devcn-ui add aspect-ratio
```

## Usage

```tsx
import { AspectRatio } from '@repo/shadcn-ui/components/aspect-ratio';

export default function Example() {
  return (
    <AspectRatio>
      Your content here
    </AspectRatio>
  );
}
```
